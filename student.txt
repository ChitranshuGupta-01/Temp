import CasePage from "@/components/CasePage";
import AnalysisPage from "@/components/AnalysisPage";

interface Props {
  params: Promise<{ type: string; id: string }>;
}

export default async function AnalysisWrapper({ params }: Props) {
  const { type, id } = await params; // âœ… await params

  if (type === "caseId") {
    return <CasePage />;
  }

  return <AnalysisPage type={type} id={id} />;
}
