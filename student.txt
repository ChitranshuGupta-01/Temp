"use client";

import React, { useEffect, useState } from "react";
import CaseDetails from "@/components/CaseDetails";
import CaseRelationDiagram from "@/components/CaseRelationDiagram";
import PastEventsLog from "@/components/PastEventsLog";
import RelatedTransactions from "@/components/RelatedTransactions";

export default function CaseDetailPage({ params }: { params: { caseId: string } }) {
  const [data, setData] = useState<any | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // fetch from public/data/caseData.json
    fetch("/data/caseData.json")
      .then((res) => res.json())
      .then((d) => setData(d))
      .finally(() => setLoading(false));
  }, []);

  if (loading) return <div className="p-12 text-center">Loading case...</div>;
  if (!data) return <div className="p-12 text-center text-red-500">Failed to load data</div>;

  return (
    <div className="min-h-screen bg-gray-50 p-6 space-y-8">
      {/* Header area */}
      <div className="max-w-7xl mx-auto">
        <CaseDetails caseData={data.case} />
      </div>

      <div className="max-w-7xl mx-auto space-y-8">
        <CaseRelationDiagram relation={data.relationDiagram} />

        <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
          <PastEventsLog events={data.pastEvents} />
          <RelatedTransactions transactions={data.transactions} />
        </div>
      </div>
    </div>
  );
}
