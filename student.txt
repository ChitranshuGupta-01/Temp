// inside Xcontroller
@PostMapping
public ResponseEntity<Xmodel> createApplication(@RequestBody Xmodel application) {
    if (application.getApplicant() == null || application.getApplicant().isEmpty() ||
        application.getHandler() == null || application.getHandler().isEmpty() ||
        application.getStatus() == null || application.getStatus().isEmpty()) {
        return ResponseEntity.badRequest().build();
    }
    return ResponseEntity.ok(service.createApplication(application));
}

@PutMapping("/{id}")
public ResponseEntity<Xmodel> updateApplication(@PathVariable String id, @RequestBody Xmodel application) {
    if (application.getApplicant() == null || application.getApplicant().isEmpty() ||
        application.getHandler() == null || application.getHandler().isEmpty() ||
        application.getStatus() == null || application.getStatus().isEmpty()) {
        return ResponseEntity.badRequest().build();
    }
    return ResponseEntity.ok(service.updateApplication(id, application));
}
