import CaseDetailPage from "@/components/CaseDetailPage";
import CasePage from "@/components/CasePage";
import AnalysisPage from "@/components/AnalysisPage";

interface Props {
  params: Promise<{ type: string; id: string }>; // ðŸ‘ˆ params is async now
}

export default async function AnalysisWrapper({ params }: Props) {
  const { type, id } = await params; // ðŸ‘ˆ await it here

  if (type === "caseId") {
    return <CaseDetailPage />;
  }

  if (type === "case") {
    return <CasePage />;
  }

  return <AnalysisPage />;
}
