package com.scb.creditcardorigination.xfeature.service;

import org.springframework.stereotype.Service;
import com.scb.creditcardorigination.xfeature.model.Xmodel;
import com.scb.creditcardorigination.xfeature.repository.Xrepository;
import java.util.List;
import java.util.Optional;

@Service
public class Xservice {
    private final Xrepository repo;
    public Xservice(Xrepository repo) { this.repo = repo; }

    public List<Xmodel> getAll() { return repo.findAll(); }
    public Optional<Xmodel> getById(String id) { return repo.findById(id); }
    public Xmodel create(Xmodel obj) { return repo.save(obj); }
    public Xmodel update(String id, Xmodel obj) {
        obj.setId(id);
        return repo.save(obj);
    }
    public void delete(String id) { repo.deleteById(id); }
}

Controller (controller/Xcontroller.java)

package com.scb.creditcardorigination.xfeature.controller;

import org.springframework.web.bind.annotation.*;
import org.springframework.http.ResponseEntity;
import org.springframework.beans.factory.annotation.Autowired;
import com.scb.creditcardorigination.xfeature.model.Xmodel;
import com.scb.creditcardorigination.xfeature.service.Xservice;
import java.util.List;
import java.util.Optional;

@CrossOrigin
@RestController
@RequestMapping("/applications")
public class Xcontroller {

    @Autowired
    private Xservice service;

    @GetMapping
    public ResponseEntity<List<Xmodel>> getAll() {
        return ResponseEntity.ok(service.getAll());
    }

    @GetMapping("/{id}")
    public ResponseEntity<Xmodel> getById(@PathVariable String id) {
        Optional<Xmodel> app = service.getById(id);
        return app.map(ResponseEntity::ok).orElseGet(() -> ResponseEntity.notFound().build());
    }

    @PostMapping
    public ResponseEntity<Xmodel> create(@RequestBody Xmodel obj) {
        return ResponseEntity.ok(service.create(obj));
    }

    @PutMapping("/{id}")
    public ResponseEntity<Xmodel> update(@PathVariable String id, @RequestBody Xmodel obj) {
        return ResponseEntity.ok(service.update(id, obj));
    }

    @DeleteMapping("/{id}")
    public ResponseEntity<String> delete(@PathVariable String id) {
        service.delete(id);
        return ResponseEntity.ok("Deleted");
    }
}

package com.scb.creditcardorigination.xfeature.exceptionHandling;

import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.http.ResponseEntity;

@ControllerAdvice
public class XexceptionHandling {
    @ExceptionHandler(Exception.class)
    public ResponseEntity<String> handleAnyException(Exception e) {
        return ResponseEntity.status(500).body("Error: " + e.getMessage());
    }
}
